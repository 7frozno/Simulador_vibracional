🎯 CARACTERÍSTICAS IMPLEMENTADAS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

✅ ANÁLISIS ESPECTRAL (FFT)
   → Transformada Rápida de Fourier
   → Identificación de componentes armónicas (1X, 2X, 3X...)
   → Espectro normalizado y filtrado
   → Resolución configurable

✅ COMPARACIÓN SANO VS DEFECTUOSO
   → Equipo sano con síntomas normales
   → Lado a lado para aprendizaje visual
   → Diferencias claras en espectro y tiempo

✅ DIAGNÓSTICO DE FALLAS (5 Tipos)
   → Desbalance (distribución no uniforme de masa)
   → Desalineamiento (ejes no coaxiales)
   → Falla de Rodamiento (defectos de bolas/pistas)
   → Cavitación (burbujas colapsando)
   → Combinada (múltiples defectos simultáneos)

✅ TENDENCIA TEMPORAL
   → RMS Móvil (Root Mean Square)
   → Pico Móvil (amplitud máxima)
   → Visualización de degradación
   → Detección de anomalías

✅ CARACTERÍSTICAS VIBRATORIAS
   → RMS (Valor eficaz)
   → Pico y Pico-Pico
   → Factor de Cresta (CF)
   → Kurtosis (Índice de impactos)
   → Asimetría (Skewness)
   → 15+ parámetros adicionales

🎓 EJEMPLO DE USO
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

from src.signal_generator import SignalGenerator, SignalConfig
from src.fault_simulator import FaultSimulator, FaultType, FaultSeverity
from src.diagnostics import DiagnosticEngine

# 1. Configurar
config = SignalConfig(rpm=1500, sampling_rate=10000, duration=1.0)

# 2. Generar falla
simulator = FaultSimulator(config)
time, signal = simulator.generate_fault(
    FaultType.UNBALANCE,
    FaultSeverity.MODERATE
)

# 3. Diagnosticar
engine = DiagnosticEngine(config.sampling_rate, config.rpm)
diagnosis = engine.diagnose(signal)

# 4. Ver resultado
print(f"Falla: {diagnosis['primary_fault'].value}")
print(f"Confianza: {diagnosis['confidence']:.1%}")
print(f"Severidad: {diagnosis['severity_level']}")
print("Acciones recomendadas:")
for accion in diagnosis['recommendations']:
    print(f"  • {accion}")
